ФУНКЦІЯ generate_pattern(n):
    # робимо масив a[n][n], заповнений "|"
    a = МАСИВ РОЗМІРУ n x n, заповнений "|"
    
    ДЛЯ i В ДІАПАЗОНІ 0 ДО n-1:
        ДЛЯ j В ДІАПАЗОНІ 0 ДО n-1:
            ЯКЩО i == j АБО (i + j) == n - 1:  # Дві діагоналі
                a[i][j] = "*"
    
    # Виводимо масив
    ДЛЯ кожного row В МАСИВІ:
        ВИВОДИ рядок, приєднуючи пробіли між символами

n = ВВІД("Введіть розмір масиву (N): ")

ЯКЩО n <= 0:
    ВИВОДИ "N повинно бути додатнім цілим числом."
ІНАКШЕ:
    generate_pattern(n)
